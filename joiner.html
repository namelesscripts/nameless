<!DOCTYPE html>
<html>
<head>
  <title>Roblox Join</title>
  <meta charset="UTF-8">
  <script>
    window.onload = function() {
      const params = new URLSearchParams(window.location.search);
      const placeId = params.get("placeId");
      const jobId = params.get("jobId");

      if (!placeId || !jobId) {
        document.body.innerHTML = "❌ placeId или jobId не указаны.";
        return;
      }

      const robloxUrl = `roblox://placeId=${placeId}&gameInstanceId=${jobId}`;
      window.location.href = robloxUrl;

      setTimeout(() => {
        const btn = document.createElement("button");
        btn.innerText = "Зайти в Roblox-сервер";
        btn.style = "font-size:20px;padding:10px 20px;";
        btn.onclick = () => {
          window.location.href = robloxUrl;
        };
        document.body.appendChild(btn);
      }, 500);
    };
  </script>
</head>
<body style="font-family:sans-serif; text-align:center; padding-top:50px;">
  <h2>⏳ Подключение к Roblox...</h2>
</body>
</html>
